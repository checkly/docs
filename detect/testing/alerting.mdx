---
title: Alerting
---
Set up notifications when checks fail:

```ts
// alert-channels.ts
import { EmailAlertChannel, SlackAlertChannel } from '@checkly/cli/constructs'

export const emailAlert = new EmailAlertChannel('team-email', {
  address: 'team@example.com',
  sendFailure: true,
  sendRecovery: true,
})

export const slackAlert = new SlackAlertChannel('team-slack', {
  url: process.env.SLACK_WEBHOOK_URL!,
  channel: '#alerts',
  sendFailure: true,
  sendRecovery: true,
})
```
